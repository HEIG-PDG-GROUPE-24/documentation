26 [Button("Update", ButtonSizes.Large)]
27 public void Update()
28 {
29 HashSet<string> visitedPairs = new();
30 // iterates over all npcs and their contacts to add the pair if it does not exist yet
31 foreach (var npcComponent in EchoesGlobal.GetAllNPCs())
32 {
33 Debug.LogFormat("Checking {0}'s contacts", npcComponent.npcData.Name);
34 foreach (var contact in npcComponent.npcData.Contacts.Where(contact => contact != null))
35 {
36 visitedPairs.Add(contact.npcData.Name + npcComponent.npcData.Name);
37 if (DistanceExists(npcComponent.npcData.Name, contact.npcData.Name)) continue;
38
39 Debug.LogFormat("Contact pair {0} - {1} did not have a distance.", contact.npcData.Name,
40 npcComponent.npcData.Name);
41 if (!_distancesBetweenContacts.ContainsKey(npcComponent.npcData.Name))
42 _distancesBetweenContacts.Add(npcComponent.npcData.Name, new Dictionary<string, double>());
43 _distancesBetweenContacts[npcComponent.npcData.Name]
44 .Add(contact.npcData.Name, (maxValue + minValue) / 2);
45 }
46 }
47
48 // any distance entry that wasn't visited should no longer exist
49 foreach (var key1 in _distancesBetweenContacts.Keys.ToList())
50 {
51 foreach (var key2 in _distancesBetweenContacts[key1].Keys.ToList().Where(key2 =>
52 !visitedPairs.Contains(key1 + key2) && !visitedPairs.Contains(key2 + key1)))
53 {
54 Debug.LogFormat("Removing pair {0} - {1}", key1, key2);
55 _distancesBetweenContacts[key1].Remove(key2);
56 if (_distancesBetweenContacts[key1].Count == 0)
57 _distancesBetweenContacts.Remove(key1);
58 }
59 }
60
61 UpdateNpcs();
62 }
