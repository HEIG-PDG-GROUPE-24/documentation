191 private double Normalize(double x, double min, double max)
192 {
193 if (x < min || x > max)
194 throw new ArgumentOutOfRangeException(nameof(x), x, "value must be between min and max inclusive");
195 return (x - min) / (max - min) * 2 - 1;

201 public double AppreciationOfPlayer()
202 {
203 double score = 0;
204 foreach (var trait in _personality.Keys)

206 double maxDiff = GlobalStats.Instance.globalTraits.maxValue -
207 GlobalStats.Instance.globalTraits.minValue;
208 double diff = Math.Abs(_opinionOfPlayer[trait] - _personality[trait]);

256 public bool ReceiveOpinion(EchoesNpcComponent from)
257 {
258 if (InPlayerInteraction && !AcceptsInterferenceDuringInteraction) return false;
259 if (!_informantsTrust.ContainsKey(from.npcData.name))
260 _informantsTrust.Add(from.npcData.name,
261 (GlobalStats.Instance.globalTraits.minValue +
262 GlobalStats.Instance.globalTraits.maxValue) / 2);
263
264 double startingPlayerAppreciation = AppreciationOfPlayer();
265
266 //adjust current opinion
267 double trustLevel = GetTrustTowards(from);
268 foreach (var trait in _opinionOfPlayer.Keys.ToList())
269 {
270 double average = (_opinionOfPlayer[trait] + from._opinionOfPlayer[trait]) / 2;
271 double diff = average - _opinionOfPlayer[trait];
272 _opinionOfPlayer[trait] += diff * ((Normalize(trustLevel,
273 GlobalStats.Instance.globalTraits.minValue,
274 GlobalStats.Instance.globalTraits.maxValue) + 1) / 2);
275 }
276
277 _lastInformantInfluence[from.npcData.name] = AppreciationOfPlayer() - startingPlayerAppreciation;
278
279 return true;
