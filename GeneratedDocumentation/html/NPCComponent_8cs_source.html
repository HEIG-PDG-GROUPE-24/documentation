254 public bool ReceiveOpinion(EchoesNpcComponent from)
255 {
256 if (InPlayerInteraction && !AcceptsInterferenceDuringInteraction) return false;
257 if (!_informantsTrust.ContainsKey(from.npcData.name))
258 _informantsTrust.Add(from.npcData.name,
259 (NPCGlobalStatsGeneratorSo.Instance.globalTraits.minValue +
260 NPCGlobalStatsGeneratorSo.Instance.globalTraits.maxValue) / 2);
261
262 double startingPlayerAppreciation = AppreciationOfPlayer();
263
264 //adjust current opinion
265 double trustLevel = GetTrustTowards(from);
266 foreach (var trait in _opinionOfPlayer.Keys.ToList())
267 {
268 double average = (_opinionOfPlayer[trait] + from._opinionOfPlayer[trait]) / 2;
269 double diff = average - _opinionOfPlayer[trait];
270 _opinionOfPlayer[trait] += diff * ((Normalize(trustLevel,
271 NPCGlobalStatsGeneratorSo.Instance.globalTraits.minValue,
272 NPCGlobalStatsGeneratorSo.Instance.globalTraits.maxValue) + 1) / 2);
273 }
274
275 _lastInformantInfluence[from.npcData.name] = AppreciationOfPlayer() - startingPlayerAppreciation;
276
277 return true;
